284  /**
285  * \brief Parse numeric, string and boolean literals
286  */
287  void parse_json_literal()
288  {
289  char c = get_skipws();
290  if ( std::isalpha(c) )
291  {
292  stream.unget();
293  std::string val = parse_json_identifier();
294  if ( val == "true" )
295  put(true);
296  else if ( val == "false" )
297  put(false);
298  else if ( val != "null" ) // Null => no value
299  put(val);
300  }
301  else if ( c == '\"' )
302  {
303  stream.unget();
304  put(parse_json_string());
305  }
306  else if ( std::isdigit(c) || c == '.' || c == '-' || c == '+' )
307  {
308  stream.unget();
309  put(parse_json_number());
310  }
311  else
312  {
313  error("Expected value");
314  }

317  /**
318  * \brief Parse a string literal
319  */
320  std::string parse_json_string()
321  {
322  char c = get_skipws();
323  std::string r;
324  if ( c == '\"' )
325  {
326  while ( true )
327  {
328  c = stream.get();
329  if ( !stream || c == '\"' )
330  break;
331  if ( c == '\\' )
332  {
333  c = escape(stream.get());
334  if ( !stream )
335  break;
336 
337  if ( c == 'u' )
338  {
339  char hex[] = "0000";
340  stream.read(hex,4);
341  for ( int i = 0; i < 4; i++ )
342  {
343  if ( !std::isxdigit(hex[i]) )
344  hex[i] = '0';
345  }
346  r += string::Utf8Parser::encode(string::to_uint(hex,16));
347 
348  continue;
349  }
350 
351  if ( !escapeable(c) )
352  r += '\\';
353  }
354 
355  r += c;
356 
357  if ( c == '\n' )
358  line++;
359  }
360  }
361 
362  return r;
363  }

365  /**
366  * \brief Parses an identifier
367  */
368  std::string parse_json_identifier()
369  {
370  char c = get_skipws();
371  if ( !std::isalpha(c) )
372  return {};
373 
374  std::string r;
375  while ( true )
376  {
377  r += c;
378  c = stream.get();
379  if ( !stream || ( !std::isalnum(c) && c != '_' && c != '-' ) )
380  break;
381  }
382  stream.unget();
383 
384  return r;
