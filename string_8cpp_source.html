32 void Utf8Parser::parse(const std::string& string)
33 {
34  input.clear();
35  input.str(string);
36  while ( true )
37  {
38  uint8_t byte = input.get();
39 
40  if ( !input )
41  break;
42 
43  // 0... .... => ASCII
44  if ( byte < 0b1000'0000 )
45  {
46  check_valid();
47  call_back(callback_ascii,byte);
48  }
49  // 11.. .... => Begin multibyte
50  else if ( (byte & 0b1100'0000) == 0b1100'0000 )
51  {
52  check_valid();
53  utf8.push_back(byte);
54 
55  // extract number of leading 1s
56  while ( byte & 0b1000'0000 )
57  {
58  length++;
59  byte <<= 1;
60  }
61 
62  // Restore byte (leading 1s have been eaten off)
63  byte >>= length;
64  unicode = byte;
65  }
66  // 10.. .... => multibyte tail
67  else if ( length > 0 )
68  {
69  utf8.push_back(byte);
70  unicode <<= 6;
71  unicode |= byte&0b0011'1111; //'
72  if ( utf8.size() == length )
73  {
74  call_back(callback_utf8,unicode,utf8);
75  unicode = 0;
76  length = 0;
77  utf8.clear();
78  }
79  }
80  }
81  check_valid();
82  call_back(callback_end);
83 }

85 std::string Utf8Parser::encode(uint32_t value)
86 {
87  if ( value < 128 )
88  return std::string(1,char(value));
89 
90  std::list<uint8_t> s;
91 
92  uint8_t head;
93  while ( value )
94  {
95  s.push_back((value&0b0011'1111)|0b1000'0000);
96  value >>= 6;
97  head <<= 1;
98  head |= 1;
99  }
100 
101  if ( (uint8_t(s.back())&0b0011'1111) > (1 << (7 - s.size())) )
102  {
103  head <<= 1;
104  head |= 1;
105  s.push_back(0);
106  }
107 
108  s.back() |= head << (8 - s.size());
109 
110  return std::string(s.rbegin(),s.rend());
111 }

302 FormattedString FormatterAnsi::decode(const std::string& source) const
303 {
304  /// \todo decode ansi codes
305  FormattedString str;
306 
307  Utf8Parser parser;
308 
309  std::string ascii;
310 
311  auto push_ascii = [&ascii,&str]()
312  {
313  if ( !ascii.empty() )
314  {
315  str.append(new AsciiSubstring(ascii));
316  ascii.clear();
317  }
318  };
319 
320  parser.callback_ascii = [&parser,&str,&ascii](uint8_t byte)
321  {
322  if ( byte == '\x1b' && parser.input.peek() == '[' )
323  {
324  parser.input.ignore();
325  std::vector<int> codes;
326  while (parser.input)
327  {
328  int i = 0;
329  if ( parser.input >> i )
330  {
331  codes.push_back(i);
332  char next = parser.input.get();
333  if ( next != ';' )
334  {
335  if ( next != 'm' )
336  parser.input.unget();
337  break;
338  }
339  }
340  }
341  FormatFlags flags;
342  bool use_flags = false;
343  for ( int i : codes )
344  {
345  if ( i == 0)
346  {
347  str.append(new ClearFormatting);
348  }
349  else if ( i == 1 )
350  {
351  flags |= FormatFlags::BOLD;
352  use_flags = true;
353  }
354  else if ( i == 22 )
355  {
356  flags &= ~FormatFlags::BOLD;
357  use_flags = true;
358  }
359  else if ( i == 4 )
360  {
361  flags |= FormatFlags::UNDERLINE;
362  use_flags = true;
363  }
364  else if ( i == 24 )
365  {
366  flags &= ~FormatFlags::UNDERLINE;
367  use_flags = true;
368  }
369  else if ( i == 39 )
370  {
371  str.append(new Color(color::nocolor));
372  }
373  else if ( i >= 30 && i <= 37 )
374  {
375  i -= 30;
376  if ( flags == FormatFlags::BOLD )
377  {
378  i &= 0xf;
379  flags = FormatFlags::NO_FORMAT;
380  use_flags = false;
381  }
382  str.append(new Color(color::Color12::from_4bit(i)));
383  }
384  else if ( i >= 90 && i <= 97 )
385  {
386  i -= 90;
387  i &= 0xf;
388  str.append(new Color(color::Color12::from_4bit(i)));
389  }
390  }
391  if ( use_flags )
392  str.append(new Format(flags));
393  }
394  else
395  {
396  ascii += byte;
397  }
398  };
399  parser.callback_utf8 = [this,&str,push_ascii](uint32_t unicode,const std::string& utf8)
400  {
401  push_ascii();
402  if ( this->utf8 )
403  str.append(new Unicode(utf8,unicode));
404  };
405  parser.callback_end = push_ascii;
406 
407  parser.parse(source);
408 
409  return str;
410 }

462 FormattedString FormatterIrc::decode(const std::string& source) const
463 {
464  FormattedString str;
465 
466  Utf8Parser parser;
467 
468  FormatFlags flags;
469  parser.callback_ascii = [&str,&flags,&parser](uint8_t byte)
470  {
471  // see https://github.com/myano/jenni/wiki/IRC-String-Formatting
472  switch ( byte )
473  {
474  case '\2':
475  if ( !(flags & FormatFlags::BOLD) )
476  {
477  flags |= FormatFlags::BOLD;
478  str.append(new Format(flags));
479  }
480  break;
481  case '\x1f':
482  if ( !(flags & FormatFlags::UNDERLINE) )
483  {
484  flags |= FormatFlags::UNDERLINE;
485  str.append(new Format(flags));
486  }
487  break;
488  case '\xf':
489  flags = FormatFlags::NO_FORMAT;
490  str.append(new ClearFormatting);
491  break;
492  case '\3':
493  {
494  std::string fg;
495  if ( std::isdigit(parser.input.peek()) )
496  {
497  fg += parser.input.get();
498  if ( std::isdigit(parser.input.peek()) )
499  fg += parser.input.get();
500 
501  if ( parser.input.peek() == ',' )
502  {
503  parser.input.ignore();
504  if ( std::isdigit(parser.input.peek()) ) parser.input.ignore();
505  if ( std::isdigit(parser.input.peek()) ) parser.input.ignore();
506  }
507  }
508  str.append(new Color(FormatterIrc::color_from_string(fg)));
509  break;
510  }
511  case '\x1d': case '\x16':
512  // Skip unsupported format flags
513  break;
514  default:
515  str.append(new Character(byte));
516  break;
517  }
518  };
519  parser.callback_utf8 = [&str](uint32_t unicode,const std::string& utf8)
520  {
521  str.append(new Unicode(utf8,unicode));
522  };
523 
524  parser.parse(source);
525 
526  return str;
527 }
